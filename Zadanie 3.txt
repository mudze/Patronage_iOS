protocol Learn{
    var taught:Bool { get set }
}

class FlashCard:Learn
{
    var taught:Bool
    
    enum TypeOfTips
    {
        case text(text: String)
        case imagine(text: String)
        case sound(text: String)
    }
    let polish:String
    let english:String
    var tips:[TypeOfTips]
    
    init(polish:String, english:String, tips:TypeOfTips... )
    {
        self.polish = polish
        self.english = english
        self.taught = false
        self.tips = tips
    }
    
    func description()->String
    {
        var returnStr = self.polish + " -> " + self.english + " ["
        
        for (index, typ) in self.tips.enumerate(){
            switch typ
            {
                case .text(let value):
                    returnStr += value
                case .imagine(let value):
                    returnStr += value
                case .sound(let value):
                    returnStr += value
                default :
                    print("something went wrong")
            }
             if index < (self.tips.count-1)
            {
                returnStr += ", "
            }
            else
            {
                returnStr += "]"
            }
        }
        return returnStr
    }
    func taughtChange()
    {
        self.taught = !self.taught
    }
}

func creationListToLearn(words: [FlashCard])->[FlashCard]
{
    var list = [FlashCard]()
    for w in words
    {
        if w.taught == false
        {
            list.append(w)
        }
    }
    return list
}
